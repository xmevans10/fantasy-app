# (urllib + csv + json) so it runs unchanged in CI and locally with just
# Python 3.11+ — no pandas/pyarrow/requests install needed. nflverse publishes
# CSV variants of every dataset, which the stdlib `csv` module reads directly.
#

pytest>=8.0
